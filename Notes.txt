1) Display list of todos

{{ todo.title }}

2) Set correct class on todos
[class.completed]="todo.completed"

3) Add trackBy
; trackBy:trackByTodos

    trackByTodos(todo: Todo) {
        return todo.id;
    }

4) Toggle completed
[(ngModel)]="todo.completed"

5) Add new todo
    private newTodoText = '';

    addTodo() {
        this.newTodoText = this.newTodoText.trim();
        if (this.newTodoText !== '') {
            this.visibleTodos.push({
                id: uuid.v4(),
                title: this.newTodoText,
                completed: false
            });
        }
        this.newTodoText = '';
    }

[(ngModel)]="newTodoText"
(ngSubmit)=addTodo()

5) Remove todo

    removeTodo(todoToRemove: Todo) {
        this.visibleTodos = this.visibleTodos.filter(
            todo => todo.id !== todoToRemove.id);
    }

 (click)="removeTodo(todo)"

6) Separate TodoItem component
a) Create todo-item.component.html
import {Component, EventEmitter, Input, Output} from '@angular/core';

import template from '/src/initial/todo-item.component.html';
import {Todo} from "./todo-list.component";

@Component({
    selector: 'todo-item',
    template: template
})
export class TodoItem {
    @Input()
    todo: Todo;

    @Output()
    onRemove = new EventEmitter<Todo>();
}

b) Copy HTML to separate file, and update:
(click)="onRemove.emit(todo)

c) Add directive to TodoList component:
directives: [TodoItem]

d)
Update todo-list.component.html
<todo-item (onRemove)="removeTodo($event)" [todo]="todo"
*ngFor="let todo of visibleTodos; trackBy:trackByTodos"></todo-item>

7) Use router
a) Add router to bootstrap
import {HashLocationStrategy, LocationStrategy} from '@angular/common';
import {ROUTER_PROVIDERS} from '@angular/router-deprecated';

bootstrap(AppComponent, [
    ROUTER_PROVIDERS,
    provide(LocationStrategy,
        {useClass: HashLocationStrategy})
]);

b)Setup routes in app.component.ts
import {RouteConfig, RouterOutlet} from '@angular/router-deprecated';

    template: '<router-outlet></router-outlet>',
    directives: [RouterOutlet]

@RouteConfig([
    {
        path: '/',
        name: 'TodoList',
        component: TodoListComponent,
